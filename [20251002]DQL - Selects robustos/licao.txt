-- Liste todos os produtos mais caros do que o produto mais barato da categoria "Eletrônicos".

select * from vendas;

select categoria, valor, produto from vendas;

select produto, valor from vendas where valor > (select min(valor) from vendas where categoria = 'Eletrônicos');

-- Mostre o vendedor que realizou a maior venda registrada na tabela.

select vendedor, produto, valor from vendas where valor = (select max(valor) from vendas);

-- Liste as categorias que têm pelo menos um produto acima do preço médio geral.

select produto, valor from vendas where valor > (select avg(valor)from vendas);
